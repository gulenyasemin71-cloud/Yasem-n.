<script>
const A = document.getElementById("videoA");
const B = document.getElementById("videoB");
const label = document.getElementById("label");
const startBtn = document.getElementById("startBtn");
const wrap = document.getElementById("wrap");
const footerText = document.getElementById("footerText");

let running = false;

startBtn.onclick = ()=>{
  A.play();
  B.play();
  A.muted = false;
  B.muted = true;
  running = true;
  startBtn.style.display = "none";

  setTimeout(()=>{
    footerText.style.opacity = 0;
    setTimeout(()=>{
      footerText.innerText =
        "Şimdi sizi aradaki farkı görmeniz için yalnız bırakıyorum.";
      footerText.style.opacity = 1;
    },500);
  },5000);
}

function fade(){
  wrap.style.setProperty("--fade","1");
  wrap.style.setProperty("opacity",".95");
  wrap.style.setProperty("filter","brightness(.6)");
  setTimeout(()=>{
    wrap.style.setProperty("opacity","1");
    wrap.style.setProperty("filter","brightness(1)");
  },200);
}

function switchCam(cam){
  if(!running){
    alert("Önce ▶ Başlat'a bas.");
    return;
  }

  const t = A.currentTime;
  A.currentTime = t;
  B.currentTime = t;

  fade();

  if(cam === "A"){
    A.style.opacity = 1;
    B.style.opacity = 0;
    A.muted = false;
    B.muted = true;
    label.innerText = "Kamera A Aktif";
  }else{
    A.style.opacity = 0;
    B.style.opacity = 1;
    A.muted = true;
    B.muted = false;
    label.innerText = "Kamera B Aktif";
  }
}

document.addEventListener("keydown",(e)=>{
  if(e.key === "a") switchCam("A");
  if(e.key === "b") switchCam("B");
});

A.onended = ()=>{
  footerText.innerText =
    "Bazen kimse fark etmez ama his çok gerçektir.";
}
</script>
